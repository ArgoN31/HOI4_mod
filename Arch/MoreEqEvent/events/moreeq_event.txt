# 合成年人事件 30

    id = techexp.30
    title = techexp.30.t
    desc = techexp.30.d
    picture = GFX_report_event_german_inspect_troops
    
    is_triggered_only = yes

    mean_time_to_happen = { days = 0 }

    immediate = {
        hidden_effect = { 
            remove_ideas = exp_clonetoman
        }
    }

    option = {
        name = techexp.30.a
        ai_chance = { factor = 1 }
        random_state = {
            limit = {
                AND= {
                    is_owned_by = ROOT
                    clonemanfactory > 0
                }
            }
            remove_building = {
                type = clonemanfactory 
                level = 1
            }
            add_manpower = 100000
        }
        random_list = {
            70 = {
                add_timed_idea = {
                    idea = clonetoman_pressure
                    days = 180
                }
            }
            30 = {
                add_manpower = 50000
            }
        }
        
    }
    option = {
        name = techexp.30.b
        ai_chance = { factor = 0 }
        add_political_power = 60
    }
}

# 纪念广场造核 33
country_event = {
    id = techexp.33
    title = techexp.33.t
    desc = techexp.33.d
    picture = GFX_report_event_german_inspect_troops
    
    is_triggered_only = yes

    mean_time_to_happen = { days = 0 }

    immediate = {
        hidden_effect = { 
            remove_ideas = eqexp_plazatocore
        }
    }

    option = {
        name = techexp.33.a
        ai_chance = { factor = 1 }
        random_list = {
            70 = {
                random_state = {
                    limit = {
                        AND= {
                            is_owned_by = ROOT
                            memorialplaza > 0
                            NOT = { is_core_of = ROOT }
                        }
                    }
                    remove_building = {
                        type = memorialplaza 
                        level = 1
                    }
                    add_core_of = ROOT
                }
            }
                
            30 = {
                 random_state = {
                    limit = {
                        AND= {
                            is_owned_by = ROOT
                            memorialplaza > 0
                            NOT = { is_core_of = ROOT }
                            NOT = { is_claimed_by = ROOT }
                        }
                    }
                    remove_building = {
                        type = memorialplaza 
                        level = 1
                    }
                    add_claim_by = ROOT
                }
            }
        }
    }
    option = {
        name = techexp.33.b
        ai_chance = { factor = 0 }
        add_political_power = 100
    }
}
